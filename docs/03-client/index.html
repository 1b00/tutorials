function useTonClient() {
  const [tonClient, setTonClient] = useState();
  useEffect(() => {
    (async () => {
      const tc = new TonClient({
        endpoint: await getHttpEndpoint(),
      });
      setTonClient(tc);
    })();
  }, []);

  return tonClient;
}
